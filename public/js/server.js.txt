const express = require('express');
const bodyParser = require('body-parser');
const path = require('path');
const MongoClient = require('mongodb').MongoClient;
const app = express();
const port = 3000;

// Serve static files (HTML, CSS, JS)
app.use(express.static(path.join(__dirname, 'public')));  // 'public' is the folder containing your static files

// Parse JSON bodies
app.use(bodyParser.json());

// Replace with your actual MongoDB URI
const uri = "mongodb+srv://4psychotic:psychotic243@cluster0.moa9o.mongodb.net/?retryWrites=true&w=majority&appName=Cluster0";
let db;

// Connect to MongoDB
MongoClient.connect(uri, { useNewUrlParser: true, useUnifiedTopology: true })
    .then(client => {
        db = client.db('<dbname>'); // Use your actual database name
        console.log('Connected to MongoDB');
    })
    .catch(error => console.error(error));

// POST route to save data
app.post('/submit', async (req, res) => {
    const { materialName, unit, materialPrice, laborPrice } = req.body;

    if (!materialName || !unit || isNaN(materialPrice) || isNaN(laborPrice)) {
        return res.status(400).json({ message: 'Invalid input' });
    }

    try {
        const collection = db.collection('daily_reports');
        const result = await collection.insertOne({
            materialName,
            unit,
            materialPrice,
            laborPrice,
            date: new Date(),
        });
        res.status(200).json({ message: 'Data saved successfully', id: result.insertedId });
    } catch (error) {
        console.error('Error saving data:', error);
        res.status(500).json({ message: 'Error saving data', error: error.message });
    }
});

// Handle GET requests to root and serve index.html
app.get('/', (req, res) => {
    res.sendFile(path.join(__dirname, 'public', 'index.html')); // Serve your HTML file
});

// Start the server
app.listen(port, () => {
    console.log(`Server running at http://localhost:${port}`);
});
